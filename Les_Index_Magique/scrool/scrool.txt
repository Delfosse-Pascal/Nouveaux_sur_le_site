<script>
    function getRandomPastelColor() {
        const r = Math.floor(Math.random() * 96 + 160); // 160-255 (pastels plus doux)
        const g = Math.floor(Math.random() * 96 + 160);
        const b = Math.floor(Math.random() * 96 + 160);
        return `rgb(${r}, ${g}, ${b})`;
    }

    function applyColorTransition(element, newColor) {
        element.style.transition = 'background-color 5s linear, color 5s linear';
        element.style.backgroundColor = newColor;
        
        if(element === document.body) {
            element.style.color = getTextColor(newColor);
        }
    }

    function getTextColor(bgColor) {
        const rgb = bgColor.match(/\d+/g);
        const brightness = (rgb[0] * 299 + rgb[1] * 587 + rgb[2] * 114) / 1000;
        return brightness > 150 ? 'rgba(20, 20, 20, 0.9)' : 'rgba(255, 255, 255, 0.95)';
    }

    function changeColors() {
        // Transition douce du fond
        applyColorTransition(document.body, getRandomPastelColor());
        
        // Transition des cellules avec décalage
        const cells = document.querySelectorAll('td');
        let delay = 0;
        
        cells.forEach(cell => {
            setTimeout(() => {
                applyColorTransition(cell, getRandomPastelColor());
            }, delay);
            delay += 200; // Décalage de 200ms entre chaque cellule
        });
    }

    // Démarrer le cycle (toutes les 15 secondes)
    changeColors();
    setInterval(changeColors, 2000);

    // Style initial
    document.head.insertAdjacentHTML('beforeend', `
        <style>
            body, td {
                transition: background-color 3s linear, color 3s linear !important;
            }
            body {
                background-color: ${getRandomPastelColor()};
                color: ${getTextColor(document.body.style.backgroundColor)};
            }
            td {
                background-color: ${getRandomPastelColor()};
            }
        </style>
    `);
</script>
