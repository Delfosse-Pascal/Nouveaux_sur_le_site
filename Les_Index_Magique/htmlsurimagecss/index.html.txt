<script>
(function() {
    const img = document.getElementById('imageMap');
    const map = document.querySelector('map[name="planetes"]');
    const areas = map.querySelectorAll('area');
    
    // Stocker les coordonnées originales
    const originalCoords = [];
    areas.forEach(area => {
        originalCoords.push(area.getAttribute('coords'));
    });
    
    function resizeMap() {
        // Attendre que l'image soit chargée pour obtenir ses dimensions naturelles
        if (!img.naturalWidth) return;
        
        const naturalWidth = img.naturalWidth;
        const naturalHeight = img.naturalHeight;
        
        // Calculer les dimensions actuelles de l'image affichée
        const windowWidth = window.innerWidth;
        const windowHeight = window.innerHeight;
        const imageRatio = naturalWidth / naturalHeight;
        const windowRatio = windowWidth / windowHeight;
        
        let displayedWidth, displayedHeight, offsetX, offsetY;
        
        if (windowRatio > imageRatio) {
            // La fenêtre est plus large que l'image
            displayedHeight = windowHeight;
            displayedWidth = displayedHeight * imageRatio;
            offsetX = (windowWidth - displayedWidth) / 2;
            offsetY = 0;
        } else {
            // La fenêtre est plus haute que l'image
            displayedWidth = windowWidth;
            displayedHeight = displayedWidth / imageRatio;
            offsetX = 0;
            offsetY = (windowHeight - displayedHeight) / 2;
        }
        
        const widthRatio = displayedWidth / naturalWidth;
        const heightRatio = displayedHeight / naturalHeight;
        
        areas.forEach((area, index) => {
            const coords = originalCoords[index].split(',');
            const newCoords = coords.map((coord, i) => {
                if (i === 0) { // x
                    return Math.round(coord * widthRatio + offsetX);
                } else if (i === 1) { // y
                    return Math.round(coord * heightRatio + offsetY);
                } else { // rayon
                    return Math.round(coord * Math.min(widthRatio, heightRatio));
                }
            });
            area.setAttribute('coords', newCoords.join(','));
        });
        
        // Repositionner l'image invisible pour la carte
        img.style.width = displayedWidth + 'px';
        img.style.height = displayedHeight + 'px';
        img.style.opacity = '0';
        img.style.pointerEvents = 'auto';
    }
    
    img.addEventListener('load', resizeMap);
    window.addEventListener('resize', resizeMap);
    
    // Déclencher au chargement
    if (img.complete) {
        resizeMap();
    }
})();
</script>
